CREATE DATABASE CompanyDB;
USE CompanyDB;

CREATE TABLE Employees (
    EmpID INT PRIMARY KEY,
    EmpName VARCHAR(100),
    EmailAddress VARCHAR(120) UNIQUE,
    CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Insert sample data
INSERT INTO Employees VALUES (1, 'John Doe', 'john@example.com', NOW());
INSERT INTO Employees VALUES (2, 'Emma Smith', 'emma@example.com', NOW());

-- Stored Procedure Example
DELIMITER $$
CREATE PROCEDURE InsertEmployee(
    IN p_id INT,
    IN p_name VARCHAR(100),
    IN p_email VARCHAR(120)
)
BEGIN
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        SELECT 'Error occurred while inserting' AS Message;
    END;

    INSERT INTO Employees(EmpID, EmpName, EmailAddress)
    VALUES (p_id, p_name, p_email);

    SELECT 'Record inserted successfully' AS Message;
END $$
DELIMITER ;

-- Select all employees
SELECT * FROM Employees;
