-- Stored procedure example
DELIMITER $$
CREATE PROCEDURE InsertEmployee(IN p_id INT, IN p_name VARCHAR(100), IN p_email VARCHAR(120))
BEGIN
  DECLARE EXIT HANDLER FOR SQLEXCEPTION
  BEGIN
    ROLLBACK;
    SELECT 'Error occurred while inserting' AS Message;
  END;

  START TRANSACTION;
  INSERT INTO Employees(EmpID, EmpName, EmailAddress) VALUES (p_id, p_name, p_email);
  COMMIT;
  SELECT 'Inserted' AS Message;
END $$
DELIMITER ;
